#include <graphics.h>
#include <iostream>
#include <vector>
#include <cmath>

using namespace std;

const int INSIDE = 0;
const int LEFT = 1;
const int RIGHT = 2;
const int BOTTOM = 4;
const int TOP = 8;

int xmin = 150, ymin = 100, xmax = 400, ymax = 300;

int computeCode(int x, int y) {
    int code = INSIDE;
    if (x < xmin) code |= LEFT;
    else if (x > xmax) code |= RIGHT;
    if (y < ymin) code |= BOTTOM;
    else if (y > ymax) code |= TOP;
    return code;
}

bool cohenSutherlandClip(int &x1, int &y1, int &x2, int &y2) {
    int code1 = computeCode(x1, y1);
    int code2 = computeCode(x2, y2);
    bool accept = false;

    while (true) {
        if ((code1 == 0) && (code2 == 0)) {
            accept = true;
            break;
        } else if (code1 & code2) {
            break;
        } else {
            int codeOut = code1 ? code1 : code2;
            int x, y;

            if (codeOut & TOP) {
                x = x1 + (x2 - x1) * (ymax - y1) / (y2 - y1);
                y = ymax;
            } else if (codeOut & BOTTOM) {
                x = x1 + (x2 - x1) * (ymin - y1) / (y2 - y1);
                y = ymin;
            } else if (codeOut & RIGHT) {
                y = y1 + (y2 - y1) * (xmax - x1) / (x2 - x1);
                x = xmax;
            } else {
                y = y1 + (y2 - y1) * (xmin - x1) / (x2 - x1);
                x = xmin;
            }

            if (codeOut == code1) {
                x1 = x; y1 = y; code1 = computeCode(x1, y1);
            } else {
                x2 = x; y2 = y; code2 = computeCode(x2, y2);
            }
        }
    }
    return accept;
}

void drawWindow() {
    setcolor(RED);
    rectangle(xmin, ymin, xmax, ymax);
}

int main() {
    int gd = DETECT, gm;
    initgraph(&gd, &gm, NULL);

    vector<pair<int, int>> polygon;

    drawWindow();

    cout << "Enter number of polygon vertices: ";
    int n;
    cin >> n;

    cout << "Enter coordinates as x y pairs:\n";
    for (int i = 0; i < n; i++) {
        int x, y;
        cin >> x >> y;
        polygon.push_back({x, y});
    }

    setcolor(WHITE);
    for (int i = 0; i < n; ++i) {
        int x1 = polygon[i].first;
        int y1 = polygon[i].second;
        int x2 = polygon[(i + 1) % n].first;
        int y2 = polygon[(i + 1) % n].second;
        line(x1, y1, x2, y2);
    }

    delay(2000);
    cleardevice();
    drawWindow();
    setcolor(GREEN);

    for (int i = 0; i < n; ++i) {
        int x1 = polygon[i].first;
        int y1 = polygon[i].second;
        int x2 = polygon[(i + 1) % n].first;
        int y2 = polygon[(i + 1) % n].second;
        if (cohenSutherlandClip(x1, y1, x2, y2)) {
            line(x1, y1, x2, y2);
        }
    }

    getch();
    closegraph();
    return 0;
}
